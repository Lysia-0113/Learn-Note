public int maxArea(int[] height) {
        int max=0;
        int right=height.length-1,left=0;
        while (right>left){
            int area=Math.min(height[left],height[right])*(right-left);
            max=Math.max(max,area);
            if(height[right]>height[left]) left++;
            else right--;
        }
        return max;
    }
